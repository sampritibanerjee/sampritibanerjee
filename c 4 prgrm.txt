//6. Print the sum of nth term of 1+11+111+..... series.

#include <stdio.h>

int main()
{
  int n,i; 
  long int sum=0,t=1;
 
  printf("input the nth term : ");
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
    
     sum=sum+t;
     t=(t*10)+1;
  }
  printf("Sum: %ld\n",sum);
  return 0;
}
